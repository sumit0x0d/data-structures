cmake_minimum_required(VERSION 3.22.1)

project(data-structures)

set(CMAKE_BUILD_TYPE Release)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_SHARED_LIBRARY_PREFIX "")

add_compile_options(-Wall -Wextra -g -Werror)
# set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -pg")

add_library(array SHARED array.c)
add_library(avl-tree SHARED avl-tree.c array.c stack.c circular-buffer.c)
add_library(binary-heap SHARED binary-heap.c)
add_library(bloom-filter SHARED bloom-filter.c array.c)
add_library(circular-buffer SHARED circular-buffer.c array.c)
add_library(doubly-linked-list SHARED doubly-linked-list.c)
add_library(dynamic-array SHARED dynamic-array.c array.c)
add_library(hash-table SHARED hash-table.c)
add_library(linked-list SHARED linked-list.c)
add_library(matrix SHARED matrix.c array.c)
add_library(red-black-tree SHARED red-black-tree.c array.c stack.c circular-buffer.c)
add_library(stack SHARED stack.c array.c)
add_library(trie SHARED trie.c)
add_library(xor-linked-list SHARED xor-linked-list.c)

target_include_directories(array PUBLIC include)
target_include_directories(avl-tree PUBLIC include)
target_include_directories(binary-heap PUBLIC include)
target_include_directories(bloom-filter PUBLIC include)
target_include_directories(circular-buffer PUBLIC include)
target_include_directories(doubly-linked-list PUBLIC include)
target_include_directories(dynamic-array PUBLIC include)
target_include_directories(linked-list PUBLIC include)
target_include_directories(hash-table PUBLIC include)
target_include_directories(matrix PUBLIC include)
target_include_directories(red-black-tree PUBLIC include)
target_include_directories(stack PUBLIC include)
target_include_directories(trie PUBLIC include)
target_include_directories(xor-linked-list PUBLIC include)
